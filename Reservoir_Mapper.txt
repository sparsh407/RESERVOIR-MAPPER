// Function to mosaic any image collection by time interval
function mosaicByTimeInterval(imageCollection, startDate, endDate, intervalDays) {
  var intervals = ee.List.sequence(0, endDate.difference(startDate, 'day'), intervalDays);
  
  var mosaickedCollection = intervals.map(function(dayOffset) {
    var start = startDate.advance(dayOffset, 'day');
    var end = start.advance(intervalDays, 'day');
    
    // Filter the image collection by the interval
    var intervalCollection = imageCollection.filterDate(start, end);
    
    // Mosaic the image collection
    var mosaickedImage = ee.Algorithms.If(
      intervalCollection.size().gt(0),
      intervalCollection.mosaic().set({
        'system:time_start': start.millis(),
        'interval_start': start.format('YYYY-MM-dd')
      }),
      null
    );
    
    return mosaickedImage;
  });
  
  // Remove null values from the final collection
  var filteredCollection = ee.ImageCollection(ee.List(mosaickedCollection).removeAll([null]));
  
  // Add the start date as a property to each image
  return filteredCollection.map(function(image) {
    var startDate = ee.Date(image.get('system:time_start')).format('YYYY-MM-dd');
    return image.set('start_date', startDate);
  });
}

// Function to calculate the effective area percentage for each image
//Note: Cloud_cover_% within the geometry is 100- effective area percentage

function calculateEffectiveAreaPercent(image) {
  // Calculate the pixel area in square meters
  var pixelArea = ee.Image.pixelArea().divide(1e6); // Convert to square kilometers
  
  // Calculate the area of valid (non-masked) pixels
  var validPixelArea = image.mask().multiply(pixelArea).reduceRegion({
    reducer: ee.Reducer.sum(),
    geometry: bufferedLake,
    scale: 10,  // Ensure it matches the scale used for the image
    maxPixels: 1e9
  }).get('B2'); // Use any valid band which has not been masked after cloud masking
  
  // Calculate the total area of the buffered lake
  var lakeArea = bufferedLake.area(10).divide(1e6); // Total area in square kilometers
  
  // Calculate the effective area percentage
  var effectiveAreaPercent = ee.Number(validPixelArea).divide(lakeArea).multiply(100);
  
  // Set the effective area percentage as a property of the image
  return image.set('effective_area_percent', effectiveAreaPercent);
}

// Otsu thresholding function for index & polarization
function otsu(histogram) {
  histogram = ee.Dictionary(histogram);
  var counts = ee.Array(histogram.get('histogram'));
  var means = ee.Array(histogram.get('bucketMeans'));
  var size = counts.length().get([0]);
  var total = counts.reduce(ee.Reducer.sum(), [0]).get([0]);
  var sum = counts.multiply(means).reduce(ee.Reducer.sum(), [0]).get([0]);
  var mean = sum.divide(total);

  var indices = ee.List.sequence(1, size);
  var betweenVar = indices.map(function(i) {
    var aCounts = counts.slice(0, 0, i);
    var aTotal = aCounts.reduce(ee.Reducer.sum(), [0]).get([0]);
    var aMeans = means.slice(0, 0, i);
    var aMean = aCounts.multiply(aMeans).reduce(ee.Reducer.sum(), [0]).get([0]).divide(aTotal);

    var bCounts = counts.slice(0, i);
    var bTotal = bCounts.reduce(ee.Reducer.sum(), [0]).get([0]);
    var bMeans = means.slice(0, i);
    var bMean = bCounts.multiply(bMeans).reduce(ee.Reducer.sum(), [0]).get([0]).divide(bTotal);

    var varBetween = aTotal.multiply(bTotal).multiply(aMean.subtract(bMean).pow(2));
    return varBetween;
  });

  return means.sort(betweenVar).get([-1]);
}

// Function to calculate AWEIsh and mask water using Otsu thresholding
var calculateAWEIsh = function(image) {
  // Calculate AWEIsh
  var awei_sh = image.expression(
    'BLUE + 2.5 * GREEN - 1.5 * (NIR + SWIR1) - 0.25 * SWIR2',
    {
      'BLUE': image.select('B2'),
      'GREEN': image.select('B3'),
      'NIR': image.select('B8'),
      'SWIR1': image.select('B11'),
      'SWIR2': image.select('B12')
    }
  ).rename('AWEIsh');
  
  // Calculate histogram and apply Otsu threshold
  var histogram = awei_sh.reduceRegion({
    reducer: ee.Reducer.histogram(),
    geometry: image.geometry(),
    scale: 10,
    maxPixels: 1e8
  }).get('AWEIsh');
  
  var threshold = ee.Number(otsu(histogram));
  var waterMask = awei_sh.gt(threshold);

  // Add AWEIsh as a band and apply the water mask
  return image.addBands(awei_sh).updateMask(waterMask);
};

// Function to generate clouded scenes or mask
function calculateCloudedScene(image) {
  // Step 1: Get the effective area mask (B2 band's mask)
  var effectiveAreaMask = image.select('B2').mask();  // Using Valid Band
  //var effectiveAreaMask = image.select('SCL').eq(_).or(image.select('SCL').eq(_)).mask(); // Using SCL Bands

  // Step 2: Create an image where the entire buffered lake is 1 (representing the whole lake area)
  var lakeMask = ee.Image.constant(1).clip(bufferedLake).mask(1); // Entire lake is 1

  // Step 3: Subtract the effective area mask from the entire lake mask to get the ineffective area
  var ineffectiveAreaMask = (lakeMask.subtract(effectiveAreaMask).selfMask()).clip(bufferedLake);  // Subtract and mask where non-zero
  
  // Step 4: Retain the initial properties of the input image
  var result = ineffectiveAreaMask.copyProperties(image, ['system:time_start', 'system:footprint', 'start_date']);
  
  // Step 5: Return the image with the ineffective area and associated properties
  return result;
}

// Function to calculate water area for each Sentinel-1 image using Otsu threshold on VV polarization after speckle filtering
function createWaterMaskS1Otsu(image) {
    // Apply median filter to reduce noise
  var image2 = image.focal_median(3, 'square', 'pixels');

  // Calculate the histogram for the 'VV' band
  var histogram = image2.reduceRegion({
    reducer: ee.Reducer.histogram(),
    geometry: bufferedLake,
    scale: 10,
    maxPixels: 1e9
  }).get('VV');

  // Calculate the Otsu threshold from the histogram
  var threshold = ee.Number(otsu(histogram));

  // Create the binary water mask (1 = water, 0 = non-water)
  var waterMask = image2.lt(threshold).rename('waterMask'); // Ensure binary format (0 or 1)
  
  var result = waterMask.updateMask(waterMask);
  
    // Copy 'system:time_start' and 'system:footprint' properties
  result = result.set('system:time_start', image.get('system:time_start'))
                 .set('system:footprint', image.get('system:footprint'))
                 .set('start_date', image.get('start_date'));

  // Return the binary water mask with associated properties
  return result;
}

function findClosestImage(image, waterMasksS1Otsu) {
  var imageDate = ee.Date(image.get('start_date')); // Get the start_date from Binary_Optical_Infected_WaterMask

  // Find the closest image in waterMasksS1Otsu
  var closest = XXXX.map(function(waterImage) {
    var diff = ee.Date(waterImage.get('start_date')).difference(imageDate, 'day').abs();
    return waterImage.set('dateDifference', diff);
  }).sort('dateDifference').first();


  // Format the original date (start_date from Binary_Optical_Infected_WaterMask) to yyyy-MM-dd
  var formattedDate = imageDate.format('yyyy-MM-dd');

  // Set the formatted image_date and reapply properties (excluding start_date)
  closest = closest.set('image_date', formattedDate);

  return closest;
}

function processImages(index) {

  var Infected_Optical_waterscene = ee.Image(Binary_Optical_Infected_WaterMask.toList(Binary_Optical_Infected_WaterMask.size()).get(index));

  // Get the first images from both collections
  var image1 = ee.Image(Closest_SAR_waterscene).unmask(0);  // Sentinel-1 binary water mask
  var image2 = ee.Image(Infected_Optical_waterscene).unmask(0);  // Sentinel-2 binary water mask
  var image3 = ee.Image(intersection_cloud_mask).unmask(0);  // Sentinel-2 derived mask

  // Define Sentinel-2 as reference for projection and scale
  var referenceImage = image2; 
  var commonProjection = referenceImage.projection(); 
  var commonScale = 10; // Sentinel-2 native resolution (10m)
  
  // Directly reproject all images to the same projection & scale
  var image11 = image1.reproject({crs: commonProjection.crs(), scale: commonScale});
  var image21 = image2.reproject({crs: commonProjection.crs(), scale: commonScale});
  var image31 = image3.reproject({crs: commonProjection.crs(), scale: commonScale});
  
  // Perform Intersection (AND) & Union (OR)
  var mergedWaterMask = image31.multiply(image11).rename('Merged_Water_Mask_Intersection'); // AND operation
  var mergedWaterMask1 = mergedWaterMask.max(image21).rename('Merged_Water_Mask_Union'); // OR operation

  // Threshold to retain only pixels with value = 1 (water pixels)
  var waterMask = mergedWaterMask1.eq(1).updateMask(mergedWaterMask1.eq(1));

  // Extract properties from closestImages (FeatureCollection)
  var closestImageFeature = ee.Feature(Binary_closest_SAR_WaterMask.toList(Binary_closest_SAR_WaterMask.size()).get(index));
  var dateDifference = closestImageFeature.get('dateDifference'); // Extract dateDifference property
  
  // Extract properties from Binary_Optical_Infected_WaterMask (ImageCollection)
  var effectiveAreaPercent = Infected_Optical_waterscene.get('effective_area_percent'); // Extract effective_area_percent property
  var startDate = Infected_Optical_waterscene.get('start_date'); // Extract start_date property
  // Compute cloud percentage as 100 - effectiveAreaPercent
  var cloudPercentage = ee.Number(100).subtract(effectiveAreaPercent);
  // Add properties to the resulting mask
  var resultMask = waterMask.set({
    'start_date': startDate,
    'cloud_percentage': cloudPercentage,
    'effective_area_percent': effectiveAreaPercent,
    'dateDifference': dateDifference


  });

  return resultMask;
}

// Function to calculate water area in square kilometers for each image
var calculateWaterArea = function(image) {
  // Define pixel area in square meters (assuming a projected coordinate system)
  var pixelArea = ee.Image.pixelArea();
  
  // Mask water pixels (assuming water pixels have a value of 1)
  var waterMask = image.eq(1);
  
  // Calculate the total area of water pixels
  var waterArea = pixelArea.updateMask(waterMask)
                           .reduceRegion({
                             reducer: ee.Reducer.sum(),
                             geometry: image.geometry(),
                             scale: 10, // Adjust based on the resolution of the raster
                             maxPixels: 1e13
                           }).getNumber('area');

  // Convert to square kilometers
  var waterAreaKm2 = waterArea.divide(1e6);

  // Retain properties from the input image
  var resultImage = image.set('water_area_km2', waterAreaKm2);
  
  // Get the property names of the input image
  var properties = image.propertyNames();
  
  // Add each property to the result image
  properties = ee.List(properties);  // Ensure it's an ee.List
  
  // Iterate over properties and set them on the result image
  properties = properties.map(function(prop) {
    return resultImage.set(prop, image.get(prop));
  });
  
  return ee.Image(resultImage);
};
